#+BEGIN_SRC emacs-lisp
  (defun helm-buffers-list ()
    (interactive)
    (helm :sources '(helm-c-source-buffers-list
                     helm-c-source-recentf
                     helm-c-source-buffer-not-found)
          :buffer "*helm buffers*" :keymap helm-c-buffer-map))
#+END_SRC
